Console.WriteLine("");
Console.WriteLine("O guardiao esta escolhendo sua jogada...");

if (escolhaComputador == 1)
{

{
    Console.WriteLine("O guardiao escolheu PAPEL!");
}
else
{
    Console.WriteLine("O guardiao escolheu TESOURA!");
}

Console.WriteLine("");

if (escolhaJogador == escolhaComputador)
{
    Console.WriteLine("Empate! Parece que os dois são igualmente fortes.");
}
else if (
(escolhaJogador == 1 && escolhaComputador == 3) ||
(escolhaJogador == 2 && escolhaComputador == 1) ||
(escolhaJogador == 3 && escolhaComputador == 2)
)

    Console.WriteLine("");

}
